//!btls: Relatively Bare-bones TLS library.
//!
//!This crate is a relatively bare-bones TLS library.
//!
//!Only pretty basic TLS features are supported, including:
//!
//! * TLS 1.2
//! * AES-128-GCM, AES-256-GCM and Chacha20-Poly1305
//! * ECDHE with NSA P-256, NSA P-384, X25519 and X448.
//! * Support for RSA (PKCS#1 and PSS) and ECDSA signatures.
//! * (Experimential) Ed25519 and Ed448 signatures.
//! * Server support for external signers (RSA, RSA-PSS, ECDSA and EdDSA).
//! * Server support for external certificate selection
//! * OSCP stapling and verification
//! * CT stapling and verification
//! * Automatic certificate selection with interlocking (server only)
//! * Certificate validation (client only)
//! * TLS Exporters
//! * SNI support
//! * ALPN support
//! * RPK support
//! * EMS support (including option to require EMS).
//! * TLS features 5 (must-staple), 18 (must-ct) and 23 (must-ems).
//! * Pinning and TLSA (client only).
//!
//!To create a client TLS session, create a [`ClientConfiguration`], set up the parameters and then call
//!`.make_session()` to get the [`ClientSession`] object, which is the client session.
//!
//!To create a server TLS session, create a [`ServerConfiguration`], set up the parameters and then call
//!`.make_session()` to get the [`ServerSession`] object, which is the server session.
//!
//!To see operations one can do on sessions (both client and server), see the [`Session`] trait.
//!
//!# Optional cargo features:
//!
//! * `gnutls-ecdsa`: Use GnuTLS library for ECDSA signing (needs GnuTLS installed). Otherwise internal ECDSA signing
//!is not supported.
//! * `enable-key-debug`: Enable `DEBUG_CRYPTO_CALCS` logging (only helpful for diagnosing interop failures due to
//!keys somehow being computed wrong).
//!
//![`ClientConfiguration`]: struct.ClientConfiguration.html
//![`ClientSession`]: struct.ClientSession.html
//![`ServerConfiguration`]: struct.ServerConfiguration.html
//![`ServerSession`]: struct.ServerSession.html
//![`Session`]: trait.Session.html

///Reason TLS connection got aborted.
///
///This enumeration lists the possible reasons TLS connection ended.
#[derive(Clone)]
pub enum AbortReason
{
	///Mutual EOF.
	///
	///The connection ended by bilaterial exchange of close_notify alerts. This is the normal way TLS
	///connection gets shut down.
	MutualEof,
	///Read error.
	///
	///The TLS connection ended in low-level read error. Textual description of error is given as
	///parameter.
	ReadError(Cow<'static, str>),
	///Write error.
	///
	///The TLS connection ended in low-level write error. Textual description of error is given as
	///parameter. This is currently not used.
	WriteError(Cow<'static, str>),
	///Received alert.
	///
	///The TLS connection ended in received fatal alert. The alert number is included.
	RecvAlert(u8),
	///Sent alert.
	///
	///The TLS connection ended in sent fatal alert. The failure that caused the alert is included.
	SentAlert(TlsFailure),
	///Handshake error.
	///
	///The TLS connection was terminated by handshake error. The failure that caused the error is included.
	HandshakeError(TlsFailure),
}

impl Error for AbortReason
{
	fn description(&self) -> &str
	{
		match self {
			&AbortReason::MutualEof => "Connection closed normally",
			&AbortReason::ReadError(_) => "Connection aborted by read error",
			&AbortReason::WriteError(_) => "Connection aborted by write error",
			&AbortReason::RecvAlert(_) => "Connection aborted by received alert",
			&AbortReason::SentAlert(_) => "Connection aborted by sent alert",
			&AbortReason::HandshakeError(_) => "Connection handshake failed",
		}
	}
}

impl Debug for AbortReason
{
	fn fmt(&self, z: &mut Formatter) -> FmtResult
	{
		match self {
			&AbortReason::MutualEof => z.write_str("MutualEof"),
			&AbortReason::ReadError(ref x) => z.write_fmt(format_args!("ReadError({})", x)),
			&AbortReason::WriteError(ref x) => z.write_fmt(format_args!("WriteError({})", x)),
			&AbortReason::RecvAlert(x) => z.write_fmt(format_args!("RecvAlert({})", x)),
			&AbortReason::SentAlert(ref y) => z.write_fmt(format_args!("SentAlert({})", y)),
			&AbortReason::HandshakeError(ref x) => z.write_fmt(format_args!("HandshakeError({})", x)),
		}
	}
}

impl Display for AbortReason
{
	fn fmt(&self, z: &mut Formatter) -> FmtResult
	{
		match self {
			&AbortReason::MutualEof => z.write_str("Connection closed normally"),
			&AbortReason::ReadError(ref x) => z.write_fmt(format_args!("Connection aborted by read \
				error: {}", x)),
			&AbortReason::WriteError(ref x) => z.write_fmt(format_args!("Connection aborted by write \
				error: {}", x)),
			&AbortReason::RecvAlert(x) => {
				let alertname = alert_name_as_string(x).unwrap_or("UNKNOWN");
				z.write_fmt(format_args!("Connection aborted by received \
					alert {}[{}]", x, alertname))
			},
			&AbortReason::SentAlert(ref y) => {
				let acode = y.get_alert_num();
				let alertname = alert_name_as_string(acode).unwrap_or("UNKNOWN");
				z.write_fmt(format_args!("Connection aborted by sent alert {}[{}]: {}", acode,
					alertname, y))
			},
			&AbortReason::HandshakeError(ref x) => z.write_fmt(format_args!("Connection handshake \
				failed: {}", x)),
		}
	}
}

///A TLS Session TX IRQ handler.
///
///This trait models TX IRQ handler, called when the session wants to write something.
pub trait TxIrqHandler
{
	///The session wants to transmit data.
	///
	///This method is called when the TLS session wants to transmit data. You should respond by arranging to
	///call `session::write_tls()` on the session as soon as practical (but not synchronously from the
	///interrupt).
	///
	///If the call to `session::write_tls()` does not write all the data it wants to write in one call, this
	///handler will be called from within `session::write_tls()` to re-arm the transmission.
	fn wants_transmit(&mut self);
}

struct TxIrqFlagReceiver(Arc<AtomicBool>);

impl TxIrqHandler for TxIrqFlagReceiver
{
	fn wants_transmit(&mut self)
	{
		self.0.store(true, AtomicOrdering::SeqCst);
	}
}

///A TLS Session TX IRQ flag.
///
///An implementation of TX IRQ handler with a one-shot flag.
///
///This is intended to be used as follows:
///
///```no-run
///let (irq_sender, irq_receiver) = TxIrqFlag::new();
///session.set_tx_irq(irq_sender);
/// //...
///if irq_receiver.poll() {
/// //...
///}
///```
#[derive(Clone)]
pub struct TxIrqFlag(Arc<AtomicBool>);

impl TxIrqFlag
{
	///Create a new flag.
	///
	///This function creates a TxIrqHandler trait object suitable to set as Tx IRQ handler, and an object that
	///can monitor if TX IRQ has occured.
	///
	///# Returns:
	///
	/// * A TxIrqHandler trait object.
	/// * Object for monitoring the state.
	pub fn new() -> (Box<TxIrqHandler+Send+Sync>, TxIrqFlag)
	{
		let x = Arc::new(AtomicBool::new(false));
		let y = TxIrqFlagReceiver(x.clone());
		let z = TxIrqFlag(x);
		(Box::new(y), z)
	}
	///Read and reset the IRQ flag.
	///
	///# Parameters:
	///
	/// * `self`: The flag.
	///
	///# Returns:
	///
	/// * True if IRQ has happened since the last call, false if not.
	pub fn poll(&self) -> bool
	{
		self.0.swap(false, AtomicOrdering::SeqCst)
	}
}

///A TLS session.
///
///This trait models an TLS session.
///
///It is implemented by [`ClientSession`] for TLS client, and [`ServerSession`] for TLS server.
///
///# Plaintext buffering:
///
///For reading, plaintext is buffered. You can use `.bytes_available()` to query how many bytes are available for
///reading in buffer (single read produces the amount of bytes requested up to at least that limit).
///
///To prevent excessive read buffering, you can use `.set_high_water_mark()` to adjust the threshold where the
///session shuts off reading of the incoming TLS stream.
///
///The read buffer can be accessed by using the impiled `Read` trait, and its methods (e.g `.read()`).
///
///For writing, plaintext is also buffered. Now there are no automatic shutoffs, but you can use the
///`.bytes_queued()` method to check how many bytes are in the buffer.
///
///The write buffer can be accessed by using the impiled `Write` trait and its methods (e.g. `.write()`).
///
///The writes are buffered to 16kB chunks. If you want to output in smaller chunks, use the `.flush()` method of
///the `Write` trait to force output to be immediately sent. Note that this does not work before the handshake
///has completed to the point that data can be sent.
///
///To end the stream, use `.send_eof()`. After this is done, no further writes are allowed. On read side, `.read()`
///returns `Ok(0)` forever on EOF. One can also query if read side is at EOF by using `.is_eof()`.
///
///# Ciphertext read/write:
///
///Ciphertext read/write happens independently in both directions:
///
///For read side, one can query if session wants to read data from socket by using `.wants_read()`. Then when the
///socket is ready, one can use `.read_tls()` to read from the socket. Exactly one `.read()` call is performed.
///
///For write side, once there is something to write, the callback registered by `.set_tx_irq()` is called. Then
///after callback has arrived, calling `.write_tls()` performs exactly one `.write()` on the socket. But do not
///do the `.write_tls()` call within the callback! Also note that if there is already a pending write, then
///calling `.set_tx_irq()` will call the new irq handler.
///
///One can call `.write_tls()` even without something to write. In this case, no `.write()` on socket happens.
///
///# Handshake:
///
///When the session is first established, the sides do a handshake. This happens as number of reads and writes during
///which no data is exchanged. One can query if this has completed by using `.handshake_completed()`.
///
///Data can be queued for sending before the handshake is complete. This data is then sent as soon as possible.
///
///# Connection closure:
///
///Upon connection termination, one can obtain the cause of termination from `.aborted_by()`. The cause can be one
///of:
///
/// * `MutualEof`: Both sides exchanged EOF signals. Connection shut down normally.
/// * `ReadError`: Connection was terminated by socket read error. The description of error is available as
///                parameter.
/// * `WriteError`: Connection was terminated by socket write error. The description of error is available as
///                 parameter.
/// * `SentAlert`: Connection was terminated by this side sending a fatal TLS alert. The alert number is included
///                as paramter.
/// * `RecvAlert`: Connection was terminated by remote side sending a fatal TLS alert. The alert number is included
///                as paramter.
/// * `HandshakeError`: The connection was terminated by error during handshaking, or due to an error in handshake
///                     protocol during running connection. The description of error is available as parameter.
///
///# ALPN:
///
///After the handshake is far enough, one can get the ALPN connection uses by using `.get_alpn()`. This method
///returns `Some(x)` if ALPN has been determined. `x` can be `None` if no ALPN has been negotiated, or `Some(y)`
///where `y` is the negotiated ALPN.
///
///# Server names:
///
///After the handshake is far enough, one can get the list of names server authenticated as by using
///`.get_server_names()`. Note that on client side, for security reasons, no results are given until server
///Finished message has been received and validated (in TLS 1.2, this won't happen until the very last step before
///the handshake completes, in TLS 1.3 it will happen a bit earlier).
///
///[`ClientSession`]: struct.ClientSession.html
///[`ServerSession`]: struct.ServerSession.html
pub trait Session: IoRead+IoWrite
{
	///Set the receive high water mark.
	///
	///Sets the receive high water mark of the session. If the number of pending unread bytes is at or above
	///the high water mark, read side will be shut down until read buffer is brought below this value.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `amount`: New high water mark.
	fn set_high_water_mark(&mut self, amount: usize);
	///Set send threshold.
	///
	///This sets the send thresohld of the session. If number of pending written bytes is greater than this, the
	///bytes are flushed, even without explicit flush call.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `amount`: New send threshold (0 enables nodelay mode).
	///
	///# Notes:
	///
	/// * If this threshold is increased past the amount of data already buffered, the change is only made
	///effective on the next write. This is to ensure that TX interrupts can never be cleared except by
	///acknowledging them via `.write_tls()`.
	fn set_send_threshold(&mut self, threshold: usize);
	///Get the number of bytes currently available for read.
	///
	///Returns the number of bytes that are waiting in the read buffer. A read is guaranteed to return lesser
	///of this and read buffer size of output.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `amount`: New high water mark.
	///
	///# Return value:
	///
	/// * Number of bytes available.
	fn bytes_available(&self) -> usize;
	///Get the number of bytes currently outstanding for write.
	///
	///Returns the number of bytes that are waiting in the write buffer (does not count bytes already encrypted
	///for sending).
	///
	///# Parameters:
	///
	/// * `self`: The session.
	///
	///# Return value:
	///
	/// * Number of bytes waiting.
	fn bytes_queued(&self) -> usize;
	///Is EOF?
	///
	///Returns if the read side is at EOF, and no further input will be produced.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	///
	///# Return value:
	///
	/// * `true` if at EOF, `false` if more input will possibly come.
	///
	///# Notes:
	///
	/// * This only covers normal close_notify. It does not cover abnormal termination.
	fn is_eof(&self) -> bool;
	///Session wants to read from stream?
	///
	///This determines if call to `read_tls()` would read from the stream or not.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	///
	///# Return value:
	///
	/// * `true` if read_tls would read from the stream, `false` if it would not.
	fn wants_read(&self) -> bool;
	///Set the TX IRQ handler.
	///
	///This sets a handler to be called when `.write_tls()` would write at least one segment (including when
	///`.write_tls() itself does not write all the data it wants to).
	///
	///If the session wants to write data at the moment this function is called, the specified handler will be
	///called synchronously.,
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `handler`: The new handler.
	fn set_tx_irq(&mut self, handler: Box<TxIrqHandler+Send>);
	///Set the application data RX handler.
	///
	///This sets a handler to be called when application data record is received. This handler is called from
	///within `.read_tls()` or `.submit_tls_record()`.
	///
	///If this handler is set, then the received data is not appended to the RX buffer, so the `Read` trait
	///methods can't be used when this handler is set.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `handler`: The new handler. If None, the existing handler is cleared.
	fn set_appdata_rx_fn(&mut self, handler: Option<Box<RxAppdataHandler+Send>>);
	///Read from stream.
	///
	///Reads TLS stream from specified source.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `stream`: The stream to read.
	///
	///# Failures:
	///
	/// * If there is low-level read error, returns the appropriate error. These errors are not fatal.
	/// * If there is fatal error in TLS connection, returns that error as TlsFailure wrapped in IoError.
	fn read_tls<R:IoRead>(&mut self, stream: &mut R) -> IoResult<()>;
	///Submit TLS record for processing.
	///
	///Submits TLS record directly for processing, bypassing all per-record buffering. When combined with
	///`.set_appdata_rx_fn()`, this allows true zero-copy processing of received application data.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `record`: The record to receive, including the headers. Used as scratchpad during the call.
	///
	///# Failures:
	///
	/// * If fatal error occurs in TLS record processing, returns that error.
	/// * If connection is not expecting data, fails with TlsFailure::NoRecordExpected. This is not fatal.
	fn submit_tls_record(&mut self, record: &mut [u8]) -> Result<(), TlsFailure>;
	///Write to stream.
	///
	///Writes TLS stream to specified sink.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `stream`: The stream to write.
	///
	///# Failures:
	///
	/// * If there is low-level write error, returns the appropriate error. These errors are not fatal.
	/// * If there is fatal error in TLS connection, returns that error as TlsFailure wrapped in IoError.
	fn write_tls<W:IoWrite>(&mut self, stream: &mut W) -> IoResult<()>;
	///Zero-latency write data into TLS stream.
	///
	///Directly compies the plaintext data from input buffer to TLS records in output buffer. This call never
	///buffers the read data.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `output`: The output buffer.
	/// * `input`: The input buffer.
	///
	///# Return value:
	///
	/// * Number of bytes that have been written to output buffer.
	/// * Number of bytes that have been read from input buffer.
	///
	///# Failures:
	///
	/// * Any fatal error in TLS processing is returned as an error.
	///
	///# Notes:
	///
	/// * If this function does not run into any fatal TLS error, but does not make any progress either, it
	///returns Ok((0, 0)), that is: Successful, 0 bytes read, 0 bytes written.
	fn zerolatency_write(&mut self, output: &mut [u8], input: &[u8]) -> Result<(usize, usize), TlsFailure>;
	///Get ESTIMATE for zero-latency write maximum size.
	///
	///This does not take all factors involved in maximum write size into account. As such, it is approximate,
	///but should be correct most of the time, if there is no data in send buffers.
	///
	///In practicular, the following factors cause it to return 0:
	///
	/// * Handshake not finished transmitting.
	/// * Any data in write buffer.
	/// * 
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `outsize`: The ouput size.
	///
	///# Returns:
	///
	/// * An estimate how much zerolatency_write() would eat output, or 0 if this can't be estimated.
	fn estimate_zerolatency_size(&self, outsize: usize) -> usize;
	///Can send data?
	///
	///
	///# Parameters:
	///
	/// * `self`: The session.
	///
	///# Returns:
	///
	/// * `true` if data can be sent, `false` if not.
	///
	///# Notes:
	///
	/// * This flag can be asserted before the handshake completes. However, one should wait until handshake
	///   completes `handshake_completed()`, before actually sending data.
	fn can_tx_data(&self) -> bool;
	///Handshake complete?
	///
	///# Parameters:
	///
	/// * `self`: The session.
	///
	///# Returns:
	///
	/// * `true` if the handshake has completed, `false` if not.
	///
	///# Notes:
	///
	/// * One should not send data before completion of the handshake. However, one can write data before that
	///   point, and such data can be sent on the wire before the handshake completes.
	fn handshake_completed(&self) -> bool;
	///Send EOF.
	///
	///Sends an EOF. After EOF has been sent, no further writes are possible. When the EOF bubbles through to
	///the wire side, it gets translated into close_notify alert.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	fn send_eof(&mut self);
	///Get error that aborted the TLS connection.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	///
	///# Returns:
	///
	/// * If the connection is still up, `None`.
	/// * Otherwise, the error that caused the session to fail.
	fn aborted_by(&self) -> Option<AbortReason>;
	///Extract keying material.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	/// * `label`: The label to use.
	/// * `context`: The context to use (`None` means no context).
	/// * `buffer`: The buffer to store to.
	///
	///# Failures:
	///
	/// * If the parameters are invalid, fails with `()`.
	/// * If the handshake isn't complete, fails with `()`.
	/// * If EMS or TLS 1.3 has not been negotiated, fails with `()`.
	fn extractor(&self, label: &str, context: Option<&[u8]>, buffer: &mut [u8]) -> Result<(), TlsFailure>;
	///Get the APLN in effect.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	///
	///# Return value:
	///
	/// * If the ALPN isn't known yet, `None`.
	/// * If the ALPN is known not to exist, `Some(None)`.
	/// * If the ALPN is known to exist, `Some(Some(alpn))`, where `alpn` is the alpn.
	fn get_alpn(&self) -> Option<Option<Arc<String>>>;
	///Get the names server authenticated as.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	///
	///# Return value:
	///
	/// * If the name list isn't known yet, `None`.
	/// * Otherwise, `Some(x)`, where `x` is the name list.
	fn get_server_names(&self) -> Option<Arc<Vec<String>>>;
	///Get the SNI client sent.
	///
	///Note that SNI should only be used for routing, and for protocols that do not have host headers on
	///application level. Do not use it with protocols that do have host headers, e.g. HTTP.
	///
	///# Parameters:
	///
	/// * `self`: The session.
	///
	///# Return value:
	///
	/// * If the name isn't known yet, `None`.
	/// * Otherwise, `Some(x)`, where `x` is the name (might be `None` if no SNI was sent).
	fn get_sni(&self) -> Option<Option<Arc<String>>>;
	///Get information about TLS connection.
	///
	///# Return value:
	///
	/// * Information about connection.
	fn connection_info(&self) -> TlsConnectionInfo;
	///Get size of first record in buffer.
	///
	///# Parameters:
	///
	/// * `self`: The session to query.
	/// * `buffer`: The buffer to inspect.
	///
	///# Return value:
	///
	/// * If the buffer has insufficient contents to even say how large the record is, `None`.
	/// * Otherwise, `Some(size)`, where `size` is the size of record (including header) in bytes.
	fn get_record_size(&self, buffer: &[u8]) -> Option<usize>;
	///Begin transactional application data read.
	///
	///This method reads from unencrypted read buffer of TLS session. The data read is not removed.
	///
	///# Parameters:
	///
	/// * `self`: The session to manipulate.
	/// * `buffer`: Buffer to store the data read to.
	///
	///# Returns:
	///
	/// * On success, buffer containing the data (sliced from `buffer`).
	/// * On failure, an error.
	///
	///# Notes:
	///
	/// * If there is no data to be read, this just succeeds with an empty buffer.
	/// * Do not start multiple transactional reads at once, it won't work.
	///
	///# Example:
	///
	///```no-run
	///let mut buf = [0u8; 16384];
	///let buf = sess.begin_transactional_read(&mut buf).unwrap();
	///let wlen = appsock.write(buf).unwrap();
	///sess.end_transactional_read(wlen).unwrap();
	///```
	fn begin_transacted_read<'a>(&self, buffer: &'a mut [u8]) -> Result<&'a [u8], TlsFailure>;
	///End transactional application data read.
	///
	///This method discards specified amount of data from unencrypted read buffer of TLS session.
	///
	///
	///# Parameters:
	///
	/// * `self`: The session to manipulate.
	/// * `len`: The number of bytes successfully processed.
	///
	///# Returns:
	///
	/// * On success, `Ok(())`.
	/// * On failure, an error.
	///
	///# Notes:
	///
	/// * See `Session::begin_transactional_read()` for an example.
	fn end_transacted_read(&mut self, len: usize) -> Result<(), TlsFailure>;
}

///IoError or TlsFailure
pub enum IoErrorOrTlsFailure
{
	///std::io::Error (can't be resolved as btls::TlsFailure)
	IoError(IoError),
	///btls::TlsFailure
	TlsFailure(TlsFailure),
}

///Take a IoResult error, and return embedded TlsFailure if any or return the original.
///
///# Parameters:
///
/// * `error': The error to inspect.
///
///# Returns:
///
/// * Embedded TlsFailure if any.
/// * Otherwise `error`.
pub fn unwrap_tls_failure(error: IoError) -> IoErrorOrTlsFailure
{
	if let Some(y) = error.get_ref() {
		if let Some(z) = y.downcast_ref::<TlsFailure>() {
			IoErrorOrTlsFailure::TlsFailure(z.clone());
		}
	}
	IoErrorOrTlsFailure::IoError(error)
}
