//!Types related to server keys.
//!
//!This module contains the various types related to server keys. This includes:
//!
//! * The base trait for server keys: `KeyPair`.
//! * Key pairs loaded from files, or file-like things: `LocalKeyPair`.
//!
//!It also contains auxillary type `SubjectPublicKeyInfo`.

use ::futures::FutureReceiver;
use ::stdlib::{Arc, Cow, SliceIter, Vec};

mod local;
pub use self::local::{KeyLoadingError, KeyReadError, LocalKeyPair, ModuleError};
#[cfg(feature = "gnutls-ecdsa")]
pub use self::local::{EccExported, GnutlsDatum, GnutlsDatumMalloc, GnutlsX509PrivkeyWrap};

///A SubjectPublicKeyInfo.
#[derive(Clone)]
pub struct SubjectPublicKeyInfo(pub Arc<Vec<u8>>);

///Supported schemes.
#[derive(Clone, Debug)]
pub struct SupportedSchemes(pub Arc<Cow<'static, [u16]>>);

impl SupportedSchemes
{
	///Forward iterator.
	pub fn iter<'a>(&'a self) -> SliceIter<'a, u16>
	{
		self.0.iter()
	}
	///Forward length.
	pub fn len(&self) -> usize
	{
		self.0.len()
	}
}

///A key pair (public key and private key).
///
///This trait is for objects that represent a keypair. It provodes signing capability with a raw
///private key and the associated public key.
pub trait KeyPair
{
	///Sign a message.
	///
	///# Parameters:
	///
	/// * `self`: The signer.
	/// * `data`: The raw data to sign.
	/// * `algorithm`: The TLS algorithm code to sign with. The following are known:
	///   * 0x0401 => RSA-PKCS1 /w SHA-256.
	///   * 0x0501 => RSA-PKCS1 /w SHA-384.
	///   * 0x0601 => RSA-PKCS1 /w SHA-512.
	///   * 0x0403 => ECDSA /w SHA-256.
	///   * 0x0503 => ECDSA /w SHA-384.
	///   * 0x0603 => ECDSA /w SHA-512.
	///   * 0x0804 => RSA-PSS /w SHA-256.
	///   * 0x0805 => RSA-PSS /w SHA-384.
	///   * 0x0806 => RSA-PSS /w SHA-512.
	///   * 0x0807 => Ed25519.
	///   * 0x0808 => Ed448.
	///
	///# Returns:
	///
	/// * A future for:
	///   * The the algorithm and actual signature, wrapped in `Ok()`.
	///   * Or on failure, `Err(())`.
	///
	///# Notes:
	///
	/// * Because signing can involve contacting an external signer, the return value is a future
	///   and not a direct value. If the future has not settled, the handshake will be paused
	///   until it is.
	fn sign(&self, data: &[u8], algorithm: u16) -> FutureReceiver<Result<Vec<u8>, ()>>;
	///Get list of TLS signature schemes this keypair is capable of signing with.
	///
	///
	///# Parameters:
	///
	/// * `self`: The signer.
	///
	///# Return value:
	///
	/// * Array of TLS signature scheme values supported.
	fn get_schemes(&self) -> SupportedSchemes;
	///
	///Get the public key.
	///
	///# Parameters:
	///
	/// * `self`: The signer.
	///
	///# Returns:
	///
	/// * The public key in X.509 SubjectPublicKeyInfo format (with the initial SEQUENCE).
	fn get_public_key(&self) -> SubjectPublicKeyInfo;
}
